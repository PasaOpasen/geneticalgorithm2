 83    @staticmethod
 84    def RemoveDuplicates(
 85        oppositor: Optional[OppositorFunc] = None,
 86        creator: Optional[CreatorFunc] = None,
 87        converter: Optional[Callable[[array1D], array1D]] = None
 88    ) -> MiddleCallbackActionFunc:
 89        """
 90        Removes duplicates from population
 91
 92        Args:
 93            oppositor: oppositor from OppOpPopInit, optional
 94                oppositor for applying after duplicates removing.
 95                None (default) means to just use the random initializer from creator.
 96            creator: the function creates population samples, optional
 97                the function creates population samples if oppositor is None. The default is None.
 98            converter: function converts (preprocesses) population samples in new format to compare (if needed)
 99                before duplicates will be searched
100
101        """
102
103        assert creator is not None or oppositor is not None, (
104            "No functions to fill population! creator or oppositors must be not None"
105        )
106
107        #
108        # create the function searches for duplicates
109        #
110        if converter is None:
111            def without_dup(pop: array2D, scores: array1D) -> Tuple[array2D, int]:
112                """returns population without dups"""
113                _, index_of_dups = np.unique(pop, axis=0, return_index=True)
114                return union_to_matrix(pop[index_of_dups], scores[index_of_dups]), pop.shape[0] - index_of_dups.size
115        else:
116            def without_dup(pop: array2D, scores: array1D) -> Tuple[array2D, int]:
117                """returns population without dups"""
118                _, index_of_dups = np.unique(
119                    np.array([converter(r) for r in pop]),
120                    axis=0, return_index=True
121                )
122                return union_to_matrix(pop[index_of_dups], scores[index_of_dups]), pop.shape[0] - index_of_dups.size
123
124        #
125        # create the function will create new samples instead of removed duplicates
126        #
127        if oppositor is None:
128            def remover(pop: array2D, scores: array1D, set_function: Callable[[array2D], array1D]) -> array2D:
129
130                pp, count_to_create = without_dup(pop, scores)  # pop without dups
131                pp2 = np.empty((count_to_create, pp.shape[1])) 
132                pp2[:, :-1] = SampleInitializers.CreateSamples(creator, count_to_create)  # new pop elements
133                pp2[:, -1] = set_function(pp2[:, :-1])  # new elements values
134
135                new_pop = np.vstack((pp, pp2))
136
137                return new_pop[new_pop[:, -1].argsort()]  # new pop
138
139        else:  # using oppositors
140            def remover(pop: array2D, scores: array1D, set_function: Callable[[array2D], array1D]) -> array2D:
141
142                pp, count_to_create = without_dup(pop, scores)  # pop without dups
143
144                if count_to_create > pp.shape[0]:
145                    raise Exception("Too many duplicates, cannot oppose")
146                
147                if count_to_create == 0:
148                    return pp[pp[:, -1].argsort()]
149                
150                pp2 = np.empty((count_to_create, pp.shape[1])) 
151                # oppose count_to_create worse elements
152                pp2[:, :-1] = OppositionOperators.Reflect(pp[-count_to_create:, :-1], oppositor)  # new pop elements
153                pp2[:, -1] = set_function(pp2[:, :-1])  # new elements values
154                    
155                new_pop = np.vstack((pp, pp2))
156                    
157                return new_pop[new_pop[:, -1].argsort()]  # new pop
158
159        def func(data: MiddleCallbackData):
160            new_pop = remover(
161                data.last_generation.variables,
162                data.last_generation.scores,
163                data.set_function
164            )
165
166            data.last_generation = Generation.from_pop_matrix(new_pop)
167
168            return data
169        
170        return func

293        def func(data: MiddleCallbackData):

303    @staticmethod
304    def ReduceMutationGen(
305        reduce_coef: float = 0.9,
306        min_mutation: float = 0.005,
307        reduce_each_generation: int = 50,
308        reload_each_generation: int = 500
309    ) -> MiddleCallbackFunc:
310
311        start_mutation = None
312
313        def func(data: MiddleCallbackData):
314            nonlocal start_mutation
315            
316            gen = data.current_generation
317            mut = data.mutation_prob
318
319            if start_mutation is None:
320                start_mutation = mut
321
322            c1 = gen % reduce_each_generation == 0
323            c2 = gen % reload_each_generation == 0
324
325            if c2:
326                mut = start_mutation
327            elif c1:
328                mut *= reduce_coef
329                mut = fast_max(mut, min_mutation)
330
331            data.mutation_prob = mut
332
333            return data, (c1 or c2)
334
335        return func

133                pp2 = np.empty((count_to_create, pp.shape[1])) 
134                pp2[:, :-1] = SampleInitializers.CreateSamples(creator, count_to_create)  # new pop elements
135                pp2[:, -1] = set_function(pp2[:, :-1])  # new elements values

142            def remover(pop: array2D, scores: array1D, set_function: Callable[[array2D], array1D]) -> array2D:
143
144                pp, count_to_create = without_dup(pop, scores)  # pop without dups
145
146                if count_to_create > pp.shape[0]:
147                    raise Exception("Too many duplicates, cannot oppose")
148

150                    return pp[pp[:, -1].argsort()]
151                
152                pp2 = np.empty((count_to_create, pp.shape[1])) 
153                # oppose count_to_create worse elements
154                pp2[:, :-1] = OppositionOperators.Reflect(pp[-count_to_create:, :-1], oppositor)  # new pop elements
155                pp2[:, -1] = set_function(pp2[:, :-1])  # new elements values

158

164                data.last_generation.scores,
165                data.set_function
